clearCache=日期 [为清除initJs.min.js的缓存 ,此处要跟着项目走，可以给个日期值来清除缓存 clearCache=日期]

行业: msDefaultPlugin


依懒文件

	

参数设置

var industryComData = [],industryDefault = [];
//industryComData = ["101","102","103","104"];	//设置常用数据
industryDefault = ["101","102"];	//设置默认值

$("#p-data-wrap1").msDefaultPlugin({
		
	type:'industry',	//类型可为 industry,skill,certificate ，必填
	check:'checkbox',	//可设置为 radio/checkbox 默认为checkbox可省略
	maxCount:5,		//多选下的最多选择条数,如为单选可删除
	column:4,		//默认为6列,列数根据字数多少来设置
	allDataMap:industryDataMap,	//配置不同的数据源,此处为行业数据 必需有。
	comData:industryComData,	//配置常用数据可删除
	init:industryDefault,	//配置默认值，无默认值可删除
	name:'industryCode_',//设置隐藏input的name属性值。
	style:{	//设置部分元素样式
		"foldHeadWidth":"120px",
		"msCbComponent":"165px",
		"msOkWidth":"250px"
	},
	align:"left"	//设置下拉对齐方式,可设置left/right不设置,默认是left
});

城市: msCityPlugin


参数设置

var cityComData=[],cityDefault=[];
	
cityComData = ["北京","天津","上海","重庆","石家庄","唐山"]; //设置常用数据
cityDefault = ["石家庄","北京"];	//设置默认值


$("#p-data-wrap").msCityPlugin({
		type:'city',	//默认为city可省略
		check:'checkbox',	//可设置为 radio/checkbox 默认为checkbox可省略
		maxCount:3,		//多选下的最多选择条数,默认为5条,如为5条可省略
		column:6,		//默认为6列此处可省
		allDataMap:cityDataMap,	//配置不同的数据源,此处为城市数据 必需有。
		comData:cityComData,	//配置常用数据可删除
		init:cityDefault,	//配置默认值，无默认值可删除
		name:'cityCode_', //设置隐藏input的name属性值。默认为cityCode,见意按需要自已设置。
		style:{		//设置部分元素样式
			"foldHeadWidth":"165px",
			"msCbComponent":"130px",
			"msOkWidth":"200px"
		},
		align:"left"	//设置下拉对齐方式,可设置left/right不设置,默认是left
});
